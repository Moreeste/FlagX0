-- Crear imagen docker
docker build -t image-flagx0 -f Dockerfile .

-- Ver imagenes instaladas
docker images

-- Crear contenedor
docker create --name flagx0 -p 8080:8080 image-flagx0

-- Ejecutar un contenedor
docker start flagx0

-- Detener un contenedor
docker stop flagx0

-- Eliminar un contenedor
docker rm flagx0

-- Eliminar imagen
docker rmi image-flagx0
